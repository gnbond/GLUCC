16/*
17 * Older MacOS versions (in particular, the clang version in MacOs XCode 10
18 * which is the latest supported on OsX 10.13) have a bug that means the usual
19 * C++17 SFINAE idiom using declval<>(), decltype() and std::void_t does not
20 * work correctly. We detect that here and use an alternate idiom for those
21 * compilers. The ugly #define is injected by the XCode version, we assume
22 * XCode 11 for OsX 10.14 and beyond are OK, but I can't test that.
23 *
24 * On these systems, `decltype(std::declval<S>() << std::declval<T>())` does not
25 * work as expected (SFINAE does not correctly propagate up to the template
26 * specialization so the resulting trait gives false positives). It works if the
27 * `std::declval<S>()` is replaced by either an actual variable (`std::cout` or
28 * even `S{}`), but this limits the trait to S types that are
29 * default-constructable.
30 *
31 * The problem appears to be related to XCode version or possibly OS version, as
32 * installing a clang-11 from Homebrew fails to fix the problem.
33 */
34#if defined(__APPLE__) && defined(__clang__) && \
35 __ENVIRONMENT_MAC_OS_X_VERSION_MIN_REQUIRED__ <= 101300
36#define BROKEN_DECLTYPE
37#endif
