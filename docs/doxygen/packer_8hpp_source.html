218 // Allow C-style arrays of any non-byte-sized type for which we already have
219 // an inserter. This uses the details: so has to be a free function
220 // template, not a member. C-style arrays of byte-sized types are handled in
221 // the class definition above using a direct call to copy() for efficiency
222 template <typename T, std::size_t N>
223 std::enable_if_t<sizeof(T) >= 2 && glucc::is_insertable_into_v<T, packer>,
224 packer&>

226 const T* p = b;
227 for (int n = N; n > 0; --n, ++p) {
228 *this << *p;
