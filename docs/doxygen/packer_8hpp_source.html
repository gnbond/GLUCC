195// True if T has a defined inserter for packer
196template <typename T>
197constexpr bool is_insertable_v =
198 std::is_reference_v<decltype(std::declval<packer>().operator<<(
199 std::declval<T>()))>;

201// True if T has size > 1 and a defined inserter for packer
202template <typename T>
203constexpr bool is_nonbyte_insertable_v = sizeof(T) > 1 && is_insertable_v<T>;

207// Allow C-style arrays of any non-byte-sized type for which we already have an
208// inserter. This uses the details: so has to be a free function template, not
209// a member. C-style arrays of byte-sized types are handled in the class
210// definition above using a direct call to copy() for efficiency
211template <typename B, std::size_t N>
212std::enable_if_t<details::is_nonbyte_insertable_v<B>, packer&> operator<<(
213 packer& out, const B (&b)[N]) {
214 const B* p = b;
215 for (int n = N; n > 0; --n, ++p) out << *p;
216 return out;
217}
